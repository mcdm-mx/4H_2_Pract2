#include <18F4520.h>
#fuses INTRC_IO, NOFCMEN, NOIESO, PUT, NOBROWNOUT, NOWDT
#fuses NOPBADEN, NOMCLR, STVREN, NOLVP, NODEBUG
#use delay(clock=16000000)
//#int_rb

void main (void){
   setup_oscillator(OSC_16MHZ);
   set_tris_A(0x00);
   set_tris_E(0x08);
   set_tris_B(0xF0); 
   set_tris_C(0xFF);
   set_tris_D(0xFF);
   //port_b_pullups(true);
   //enable_interrupts(int_RB);
   //enable_interrupts(global);
   int16 resultado;
   for(;;){
      int8 primerOperando=input_c();
      int8 segundoOperando=input_d();
      if(input(pin_B4)){
         delay_ms(250);
         while((input(PIN_B4)));//efecto antirebote
         resultado=primerOperando+segundoOperando;
      }
      if(input(pin_B5)){
         delay_ms(250);
         while((input(PIN_B5)));
         resultado=primerOperando-segundoOperando;
      }
      if(input(pin_B6)){
         delay_ms(250);
         while((input(PIN_B6)));
         resultado=primerOperando*segundoOperando;
      }
      if(input(pin_B7)){
         delay_ms(250);
         while((input(PIN_B7)));

      output_B(res<<6);
      output_E(res<<10)
   }
}
